37 void maybe_process_next_entry()
38 {
39 if (stopping || queued.empty() || running.size() >= max_processes)
40 {
41 return;
42 }
43 auto entry = std::move(queued.front());
44 queued.pop();
45 handle_entry(std::move(entry));
46 }

48 void handle_entry(QueueEntry&& entry)
49 {
50 auto now = std::chrono::steady_clock::now();
51 auto queue_time_ms =
52 std::chrono::duration_cast<std::chrono::milliseconds>(
53 now - entry.queued_at)
54 .count();
55
56 auto& msg = entry.msg;
57 auto& args = msg.args;
58
59 std::vector<const char*> argv;
60 for (size_t i = 0; i < args.size(); i++)
61 {
62 argv.push_back(args.at(i).c_str());
63 }
64 argv.push_back(nullptr);
65
66 auto handle = new uv_process_t;
67 handle->data = this;
68
69 uv_process_options_t options = {};
70 options.file = argv.at(0);
71 options.args = const_cast<char**>(argv.data());
72 options.exit_cb = ProcessLauncher::on_process_exit;
73
74 auto rc = uv_spawn(uv_default_loop(), handle, &options);
75
76 if (rc != 0)
77 {
78 LOG_FAIL_FMT("Error starting host process: {}", uv_strerror(rc));
79 return;
80 }
81
82 LOG_DEBUG_FMT(
83 "Launching host process: pid={} queuetime={}ms cmd={}",
84 handle->pid,
85 queue_time_ms,
86 fmt::join(args, " "));
87
88 auto started_at = std::chrono::steady_clock::now();
89 ProcessEntry process_entry{std::move(entry.msg), started_at};
90 running.insert({handle->pid, std::move(process_entry)});
91 }

100 void on_process_exit(uv_process_t* handle, int64_t exit_status)
101 {
102 auto& process = running.at(handle->pid);
103
104 auto t_end = std::chrono::steady_clock::now();
105 auto runtime_ms = std::chrono::duration_cast<std::chrono::milliseconds>(
106 t_end - process.started_at)
107 .count();
108
109 LOG_DEBUG_FMT(
110 "Host process exited: pid={} status={} runtime={}ms cmd={}",
111 handle->pid,
112 exit_status,
113 runtime_ms,
114 fmt::join(process.msg.args, " "));
115
116 running.erase(handle->pid);
117
118 maybe_process_next_entry();
119
120 uv_close((uv_handle_t*)handle, ProcessLauncher::on_close);
121 }

123 static void on_close(uv_handle_t* handle)
124 {
125 delete handle;
126 }
