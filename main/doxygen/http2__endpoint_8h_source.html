14 class HTTP2Endpoint : public ccf::TLSEndpoint
15 {
16 protected:
17 http2::Session& session;
18
19 HTTP2Endpoint(
20 http2::Session& session_,
21 int64_t session_id,
22 ringbuffer::AbstractWriterFactory& writer_factory,
23 std::unique_ptr<tls::Context> ctx) :
24 TLSEndpoint(session_id, writer_factory, std::move(ctx)),
25 session(session_)
26 {}
27
28 public:
29 static void recv_cb(std::unique_ptr<threading::Tmsg<SendRecvMsg>> msg)
30 {
31 reinterpret_cast<HTTP2Endpoint*>(msg->data.self.get())
32 ->recv_(msg->data.data.data(), msg->data.data.size());
33 }
34
35 void recv(const uint8_t* data, size_t size, sockaddr) override
36 {
37 auto msg = std::make_unique<threading::Tmsg<SendRecvMsg>>(&recv_cb);
38 msg->data.self = this->shared_from_this();
39 msg->data.data.assign(data, data + size);
40
41 threading::ThreadMessaging::thread_messaging.add_task(
42 execution_thread, std::move(msg));
43 }
44
45 void recv_(const uint8_t* data_, size_t size_)
46 {
47 recv_buffered(data_, size_);
48
49 LOG_TRACE_FMT("recv called with {} bytes", size_);
50
51 constexpr auto read_block_size = 4096;
52 std::vector<uint8_t> buf(read_block_size);
53 auto data = buf.data();
54 auto n_read = read(data, buf.size(), false);
55
56 while (true)
57 {
58 if (n_read == 0)
59 {
60 return;
61 }
62
63 LOG_TRACE_FMT("Going to parse {} bytes", n_read);
64
65 try
66 {
67 session.recv(data, n_read);
68
69 // Used all provided bytes - check if more are available
70 n_read = read(buf.data(), buf.size(), false);
71 }
72 catch (const std::exception& e)
73 {
74 LOG_FAIL_FMT("Error parsing HTTP request");
75 LOG_DEBUG_FMT("Error parsing HTTP request: {}", e.what());
76
77 auto response = http::Response(HTTP_STATUS_BAD_REQUEST);
78 response.set_header(
79 http::headers::CONTENT_TYPE, http::headervalues::contenttype::TEXT);
80 auto body = fmt::format(
81 "Unable to parse data as a HTTP request. Error details are "
82 "below.\n\n{}",
83 e.what());
84 response.set_body((const uint8_t*)body.data(), body.size());
85 send_raw(response.build_response());
86
87 close();
88 break;
89 }
90 }
91 }
92 };

130 void handle_request(
131 llhttp_method verb,
132 const std::string_view& url,
133 http::HeaderMap&& headers,
134 std::vector<uint8_t>&& body,
135 int32_t stream_id) override
136 {
137 LOG_TRACE_FMT(
138 "Processing msg({}, {} [{} bytes])",
139 llhttp_method_name(verb),
140 url,
141 body.size());
142
143 try
144 {
145 if (session_ctx == nullptr)
146 {
147 session_ctx = std::make_shared<ccf::SessionContext>(
148 session_id, peer_cert(), interface_id);
149 }
150
151 std::shared_ptr<http::HttpRpcContext> rpc_ctx = nullptr;
152 try
153 {
154 rpc_ctx = std::make_shared<HttpRpcContext>(
155 session_ctx, verb, url, std::move(headers), std::move(body));
156 }
157 catch (std::exception& e)
158 {
159 // Note: return HTTP_STATUS_INTERNAL_SERVER_ERROR, e.what()
160 }
161
162 const auto actor_opt = http::extract_actor(*rpc_ctx);
163 if (!actor_opt.has_value())
164 {
165 rpc_ctx->set_error(
166 HTTP_STATUS_NOT_FOUND,
167 ccf::errors::ResourceNotFound,
168 fmt::format(
169 "Request path must contain '/[actor]/[method]'. Unable to parse "
170 "'{}'.",
171 rpc_ctx->get_method()));
172 // Note: return HTTP_STATUS_NOT_FOUND, e.what()
173 return;
174 }
175
176 const auto& actor_s = actor_opt.value();
177 auto actor = rpc_map->resolve(actor_s);
178 auto search = rpc_map->find(actor);
179 if (actor == ccf::ActorsType::unknown || !search.has_value())
180 {
181 rpc_ctx->set_error(
182 HTTP_STATUS_NOT_FOUND,
183 ccf::errors::ResourceNotFound,
184 fmt::format("Unknown actor '{}'.", actor_s));
185 // Note: return HTTP_STATUS_NOT_FOUND, e.what()
186 return;
187 }
188
189 auto response = search.value()->process(rpc_ctx);
190
191 if (!response.has_value())
192 {
193 // If the RPC is pending, hold the connection.
194 LOG_TRACE_FMT("Pending");
195 return;
196 }
197 else
198 {
199 server_session.send_response(
200 stream_id,
201 rpc_ctx->get_response_http_status(),
202 rpc_ctx->get_response_headers(),
203 std::move(rpc_ctx->get_response_body()));
204 }
205 }
206 catch (const std::exception& e)
207 {
208 // Note: return HTTP_STATUS_INTERNAL_SERVER_ERROR, e.what()
209 send_raw(http::error(
210 HTTP_STATUS_INTERNAL_SERVER_ERROR,
211 ccf::errors::InternalError,
212 fmt::format("Exception: {}", e.what())));
213
214 // On any exception, close the connection.
215 LOG_FAIL_FMT("Closing connection");
216 LOG_DEBUG_FMT("Closing connection due to exception: {}", e.what());
217 close();
218 throw;
219 }
220 }
