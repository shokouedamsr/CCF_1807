76 template <typename T>
77 bool get_path_param(
78 const ccf::PathParams& params,
79 const std::string& param_name,
80 T& value,
81 std::string& error)
82 {
83 const auto it = params.find(param_name);
84 if (it == params.end())
85 {
86 error = fmt::format("No parameter named '{}' in path", param_name);
87 return false;
88 }
89
90 const auto param_s = it->second;
91 const auto [p, ec] =
92 std::from_chars(param_s.data(), param_s.data() + param_s.size(), value);
93 if (ec != std::errc())
94 {
95 error = fmt::format(
96 "Unable to parse path parameter '{}' as a {}", param_s, param_name);
97 return false;
98 }
99
100 return true;
101 }
