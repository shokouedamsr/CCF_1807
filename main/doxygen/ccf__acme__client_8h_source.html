62 virtual void check_expiry(
63 std::shared_ptr<kv::Store> tables,
64 std::unique_ptr<NetworkIdentity>& identity)
65 {
66 auto now = std::chrono::system_clock::now();
67 bool renew = false;
68 auto tx = tables->create_read_only_tx();
69 auto certs = tx.ro<ACMECertificates>(Tables::ACME_CERTIFICATES);
70 auto cert = certs->get(config_name);
71 if (cert)
72 {
73 auto v = crypto::make_verifier(*cert);
74 double rem_pct = v->remaining_percentage(now);
75 LOG_TRACE_FMT(
76 "ACME: remaining certificate for '{}' validity: {}%, "
77 "{} "
78 "seconds",
79 config_name,
80 100.0 * rem_pct,
81 v->remaining_seconds(now));
82 renew = rem_pct < 0.33;
83 }
84
85 if (renew || !cert)
86 {
87 get_certificate(make_key_pair(identity->priv_key));
88 }
89 }

141 virtual void on_certificate(const std::string& certificate) override
142 {
143 // Write the endorsed certificate to the certificate table; all nodes
144 // will install it later, in the global hook on that table.
145 auto tx = store->create_tx();
146 auto certs = tx.rw<ACMECertificates>(Tables::ACME_CERTIFICATES);
147 certs->put(config_name, crypto::Pem(certificate));
148 tx.commit();
149 }
