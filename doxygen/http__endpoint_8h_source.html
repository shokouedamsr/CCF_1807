32 static void recv_cb(std::unique_ptr<threading::Tmsg<SendRecvMsg>> msg)
33 {
34 reinterpret_cast<HTTPEndpoint*>(msg->data.self.get())
35 ->recv_(msg->data.data.data(), msg->data.data.size());

38 void recv(const uint8_t* data, size_t size, sockaddr) override
39 {
40 auto msg = std::make_unique<threading::Tmsg<SendRecvMsg>>(&recv_cb);
41 msg->data.self = this->shared_from_this();
42 msg->data.data.assign(data, data + size);
43
44 threading::ThreadMessaging::thread_messaging.add_task(
45 execution_thread, std::move(msg));
46 }
47
48 void recv_(const uint8_t* data_, size_t size_)
49 {
50 recv_buffered(data_, size_);
51
52 LOG_TRACE_FMT("recv called with {} bytes", size_);
53
54 constexpr auto read_block_size = 4096;
55 std::vector<uint8_t> buf(read_block_size);
56 auto data = buf.data();
57 auto n_read = read(data, buf.size(), false);
58
59 while (true)
60 {
61 if (n_read == 0)
62 {
63 return;
64 }
65
66 LOG_TRACE_FMT("Going to parse {} bytes", n_read);
67
68 try
69 {
70 p.execute(data, n_read);
71
72 // Used all provided bytes - check if more are available
73 n_read = read(buf.data(), buf.size(), false);
74 }
75 catch (RequestPayloadTooLarge& e)
76 {
77 if (error_reporter)
78 {
79 error_reporter->report_request_payload_too_large_error(session_id);
80 }
81
82 LOG_DEBUG_FMT("Request is too large: {}", e.what());
83
84 send_raw(http::error(ccf::ErrorDetails{
85 HTTP_STATUS_PAYLOAD_TOO_LARGE,
86 ccf::errors::RequestBodyTooLarge,
87 e.what()}));
88
89 close();
90 break;
91 }
92 catch (RequestHeaderTooLarge& e)
93 {
94 if (error_reporter)
95 {
96 error_reporter->report_request_header_too_large_error(session_id);
97 }
98
99 LOG_DEBUG_FMT("Request header is too large: {}", e.what());
100
101 send_raw(http::error(ccf::ErrorDetails{
102 HTTP_STATUS_REQUEST_HEADER_FIELDS_TOO_LARGE,
103 ccf::errors::RequestHeaderTooLarge,
104 e.what()}));
105
106 close();
107 break;
108 }
109 catch (const std::exception& e)
110 {
111 if (error_reporter)
112 {
113 error_reporter->report_parsing_error(session_id);
114 }
115 LOG_DEBUG_FMT("Error parsing HTTP request: {}", e.what());
116
117 auto response = http::Response(HTTP_STATUS_BAD_REQUEST);
118 response.set_header(
119 http::headers::CONTENT_TYPE, http::headervalues::contenttype::TEXT);
120 // NB: Avoid formatting input data a string, as it may contain null
121 // bytes. Instead insert it at the end of this message, verbatim
122 auto body_s = fmt::format(
123 "Unable to parse data as a HTTP request. Error message is: {}\n"
124 "Error occurred while parsing fragment:\n",
125 e.what());
126 std::vector<uint8_t> response_body(
127 std::begin(body_s), std::end(body_s));
128 response_body.insert(response_body.end(), data, data + n_read);
129 response.set_body(response_body.data(), response_body.size());
130 send_raw(response.build_response());
131
132 close();
133 break;
134 }
135 }
136 }

176 void handle_request(
177 llhttp_method verb,
178 const std::string_view& url,
179 http::HeaderMap&& headers,
180 std::vector<uint8_t>&& body,
181 int32_t) override
182 {
183 LOG_TRACE_FMT(
184 "Processing msg({}, {} [{} bytes])",
185 llhttp_method_name(verb),
186 url,
187 body.size());
188
189 try
190 {
191 if (session_ctx == nullptr)
192 {
193 session_ctx = std::make_shared<ccf::SessionContext>(
194 session_id, peer_cert(), interface_id);
195 }
196
197 std::shared_ptr<http::HttpRpcContext> rpc_ctx = nullptr;
198 try
199 {
200 rpc_ctx = std::make_shared<HttpRpcContext>(
201 session_ctx, verb, url, std::move(headers), std::move(body));
202 }
203 catch (std::exception& e)
204 {
205 send_raw(http::error(
206 HTTP_STATUS_INTERNAL_SERVER_ERROR,
207 ccf::errors::InternalError,
208 e.what()));
209 }
210
211 const auto actor_opt = http::extract_actor(*rpc_ctx);
212 if (!actor_opt.has_value())
213 {
214 rpc_ctx->set_error(
215 HTTP_STATUS_NOT_FOUND,
216 ccf::errors::ResourceNotFound,
217 fmt::format(
218 "Request path must contain '/[actor]/[method]'. Unable to parse "
219 "'{}'.",
220 rpc_ctx->get_method()));
221 send_raw(rpc_ctx->serialise_response());
222 return;
223 }
224
225 const auto& actor_s = actor_opt.value();
226 auto actor = rpc_map->resolve(actor_s);
227 auto search = rpc_map->find(actor);
228 if (actor == ccf::ActorsType::unknown || !search.has_value())
229 {
230 rpc_ctx->set_error(
231 HTTP_STATUS_NOT_FOUND,
232 ccf::errors::ResourceNotFound,
233 fmt::format("Unknown actor '{}'.", actor_s));
234 send_raw(rpc_ctx->serialise_response());
235 return;
236 }
237
238 auto response = search.value()->process(rpc_ctx);
239
240 if (!response.has_value())
241 {
242 // If the RPC is pending, hold the connection.
243 LOG_TRACE_FMT("Pending");
244 return;
245 }
246 else
247 {
248 send_buffered(response.value());
249 flush();
250 }
251 }
252 catch (const std::exception& e)
253 {
254 send_raw(http::error(
255 HTTP_STATUS_INTERNAL_SERVER_ERROR,
256 ccf::errors::InternalError,
257 fmt::format("Exception: {}", e.what())));
258
259 // On any exception, close the connection.
260 LOG_FAIL_FMT("Closing connection");
261 LOG_DEBUG_FMT("Closing connection due to exception: {}", e.what());
262 close();
263 throw;
264 }
265 }
