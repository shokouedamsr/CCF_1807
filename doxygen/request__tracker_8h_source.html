33 struct RequestComp
34 {
35 bool operator()(const Request* lhs, const Request* rhs) const
36 {
37 const std::array<uint64_t, 4>& lhs_hash =
38 (std::array<uint64_t, 4>&)lhs->hash;
39 const std::array<uint64_t, 4>& rhs_hash =
40 (std::array<uint64_t, 4>&)rhs->hash;
41
42 for (uint32_t i = 0; i < 4; ++i)
43 {
44 if (lhs_hash[i] == rhs_hash[i])
45 {
46 continue;
47 }
48 return lhs_hash[i] > rhs_hash[i];
49 }
50 return false;
51 }
52 };

101 void tick(std::chrono::milliseconds current_time)
102 {
103 std::unique_lock<ccf::Mutex> guard(lock);
104 if (current_time < retail_unmatched_deleted_hashes)

184 static void insert(
185 const crypto::Sha256Hash& hash,
186 std::chrono::milliseconds time,
187 std::multiset<Request*, RequestComp>& requests_,
188 snmalloc::DLList<Request, std::nullptr_t, true>& requests_list_)
189 {
190 if (
191 requests_list_.get_tail() != nullptr &&
192 requests_list_.get_tail()->time > time)
193 {
194 // Time is not a precise measurement and can be different
195 // on different threads. To ensure that the time values are
196 // correctly ordered reuse the time value of the last
197 // inserted item if the new item's time is not the highest known value.
198 time = requests_list_.get_tail()->time;
199 }
200 auto r = std::make_unique<Request>(hash, time);
201 requests_.insert(r.get());
202 requests_list_.insert_back(r.release());
203 }
204
205 bool remove(
206 const crypto::Sha256Hash& hash,
207 std::multiset<Request*, RequestComp>& requests_,
208 snmalloc::DLList<Request, std::nullptr_t, true>& requests_list_)
209 {
210 Request r(hash);
211 auto ret = requests_.equal_range(&r);
212 if (ret.first == ret.second)
213 {
214 return false;
215 }
216
217 auto it = ret.first;
218 for (auto c = ret.first; c != ret.second; ++c)
219 {
220 if ((*it)->time > (*c)->time)
221 {
222 it = c;
223 }
224 }
225
226 std::unique_ptr<Request> req(*it);
227 requests_.erase(it);
228 requests_list_.remove(req.get());
229 return true;
230 }
