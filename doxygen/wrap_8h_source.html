552 class JSWrappedPropertyEnum
553 {
554 public:
555 JSWrappedPropertyEnum(JSContext* ctx, const JSWrappedValue& value)
556 {
557 if (!JS_IsObject(value))
558 {
559 throw std::logic_error(
560 fmt::format("object value required for property enum"));
561 }
562
563 JSPropertyEnum* prop_enum;
564 uint32_t prop_count;
565
566 if (
567 JS_GetOwnPropertyNames(
568 ctx,
569 &prop_enum,
570 &prop_count,
571 value,
572 JS_GPN_STRING_MASK | JS_GPN_ENUM_ONLY) == -1)
573 {
574 throw std::logic_error(
575 fmt::format("Could not extract property names of enum"));
576 }
577 for (size_t i = 0; i < prop_count; i++)
578 properties.push_back(JSWrappedAtom(ctx, prop_enum[i].atom));
579 for (uint32_t i = 0; i < prop_count; i++)
580 JS_FreeAtom(ctx, prop_enum[i].atom);
581 js_free(ctx, prop_enum);
582 }
583 ~JSWrappedPropertyEnum() {}
584
585 JSWrappedAtom operator[](size_t i) const
586 {
587 return properties[i];
588 }
589
590 size_t size() const
591 {
592 return properties.size();
593 }
594
595 JSContext* ctx;
596 std::vector<JSWrappedAtom> properties;
597 };
