20// `It is recommended that applications attempt 10 retries in a tight loop in
21// the unlikely event that the RDRAND instruction does not return a random
22// number. This number is based on a binomial probability argument: given
23// the design margins of the DRNG, the odds of ten failures in a row are
24// astronomically small and would in fact be an indication of a larger CPU
25// issue.`
26#define RDRAND_RETRIES 10

97 /* So we don't call cpuid multiple times for the same information */

109 if ((info.ecx & 0x40000000) == 0x40000000)

112 cpuid(&info, 7, 0);

187 static unsigned int rdrand_get_bytes(unsigned int n, unsigned char* dest)
188 {
189 unsigned char *headstart, *tailstart = nullptr;
190 uint64_t* blockstart;
191 unsigned int count, ltail, lhead, lblock;
192 uint64_t i, temprand;
193
194 /* Get the address of the first 64-bit aligned block in the
195 * destination buffer. */
196 headstart = dest;
197 if (((uint64_t)headstart % (uint64_t)8) == 0)
198 {
199 blockstart = (uint64_t*)headstart;
200 lblock = n;
201 lhead = 0;
202 }
203 else
204 {
205 blockstart =
206 (uint64_t*)(((uint64_t)headstart & ~(uint64_t)7) + (uint64_t)8);
207 lhead = (unsigned int)((uint64_t)blockstart - (uint64_t)headstart);
208 lblock =
209 ((n - lhead) & ~(unsigned int)7); // cwinter: this bit is/as buggy in
210 // the Intel examples.
211 }
212
213 /* Compute the number of 64-bit blocks and the remaining number
214 * of bytes (the tail) */
215 ltail = n - lblock - lhead;
216 count = lblock / 8; /* The number 64-bit rands needed */
217
218 assert(lhead < 8);
219 assert(lblock <= n);
220 assert(ltail < 8);
221
222 if (ltail)
223 tailstart = (unsigned char*)((uint64_t)blockstart + (uint64_t)lblock);
224
225 /* Populate the starting, mis-aligned section (the head) */
226 if (lhead)
227 {
228 if (!rdrand64_retry(RDRAND_RETRIES, &temprand))
229 return 0;
230 memcpy(headstart, &temprand, lhead);
231 }
232
233 /* Populate the central, aligned block */
234 for (i = 0; i < count; ++i, ++blockstart)
235 {
236 if (!rdrand64_retry(RDRAND_RETRIES, blockstart))
237 return i * 8 + lhead;
238 }
239
240 /* Populate the tail */
241 if (ltail)
242 {
243 if (!rdrand64_retry(RDRAND_RETRIES, &temprand))
244 return count * 8 + lhead;
245 memcpy(tailstart, &temprand, ltail);
246 }
247
248 return n;
249 }
