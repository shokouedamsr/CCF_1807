98 static inline std::vector<std::string_view> split(
99 const std::string_view& s,
100 const std::string_view& separator = " ",
101 size_t max_split = SIZE_MAX)
102 {
103 std::vector<std::string_view> result;
104
105 auto separator_end = 0;
106 auto next_separator_start = s.find(separator);
107 while (next_separator_start != std::string_view::npos &&
108 result.size() < max_split)
109 {
110 result.push_back(
111 s.substr(separator_end, next_separator_start - separator_end));
112
113 separator_end = next_separator_start + separator.size();
114 next_separator_start = s.find(separator, separator_end);
115 }
116
117 result.push_back(s.substr(separator_end));
118 return result;
119 }

124 static inline std::tuple<std::string_view, std::string_view> split_1(
125 const std::string_view& s, const std::string_view& separator)
126 {
127 const auto v = split(s, separator, 1);
