208 using SeqNoRange = std::pair<ccf::SeqNo, ccf::SeqNo>;
209
210 // Keep as many existing entries as possible, return indices that weren't
211 // already present to indicate they should be fetched. For example, if we
212 // were previously fetching:
213 // 2 3 4 5
214 // and then we adjust to:
215 // 4 5
216 // we don't need to fetch anything new; this is a subrange. But if we
217 // adjust to:
218 // 0 1 2 3 4 5 6
219 // we need to start fetching 0, 1, and 6.
220 SeqNoCollection adjust_ranges(
221 const SeqNoCollection& new_seqnos, bool should_include_receipts)
222 {
223 HISTORICAL_LOG(
224 "Adjusting ranges, previously {}, new {} ({} vs {})",
225 requested_seqnos.size(),
226 new_seqnos.size(),
227 include_receipts,
228 should_include_receipts);
229 if (
230 new_seqnos == requested_seqnos &&
231 should_include_receipts == include_receipts)
232 {
233 // This is precisely the request we're already tracking - do nothing
234 HISTORICAL_LOG("Already have this range");
235 return {};
236 }
237
238 std::set<SeqNo> newly_requested;
239 std::map<ccf::SeqNo, StoreDetailsPtr> new_stores;
240
241 for (auto seqno : new_seqnos)
242 {
243 auto existing_details = get_store_details(seqno);
244 if (existing_details == nullptr)
245 {
246 newly_requested.insert(seqno);
247 new_stores[seqno] = std::make_shared<StoreDetails>();
248 HISTORICAL_LOG("{} is new", seqno);
249 }
250 else
251 {
252 new_stores[seqno] = std::move(existing_details);
253 HISTORICAL_LOG("Found {} already", seqno);
254 }
255 }
256
257 requested_stores = std::move(new_stores);
258 first_requested_seqno = new_seqnos.front();
259
260 // If the range has changed, forget what ledger secrets we may have been
261 // fetching - the caller can begin asking for them again
262 ledger_secret_recovery_info = nullptr;
263
264 const auto newly_requested_receipts =
265 should_include_receipts && !include_receipts;
266
267 requested_seqnos = new_seqnos;
268 include_receipts = should_include_receipts;
269
270 HISTORICAL_LOG(
271 "Clearing {} supporting signatures", supporting_signatures.size());
272 supporting_signatures.clear();
273 if (newly_requested_receipts)
274 {
275 // If requesting signatures, populate receipts for each entry that we
276 // already have. Normally this would be done when each entry was
277 // received, but in the case that we have the entries already and only
278 // request signatures now, we delay that work to now.
279
280 for (auto seqno : new_seqnos)
281 {
282 const auto next_seqno = populate_receipts(seqno);
283 if (next_seqno.has_value())
284 {
285 newly_requested.insert(*next_seqno);
286 supporting_signatures[*next_seqno] =
287 std::make_shared<StoreDetails>();
288 }
289 }
290 }
291
292 return SeqNoCollection(newly_requested.begin(), newly_requested.end());
293 }

295 enum class PopulateReceiptsResult
296 {
297 // Common result. The new seqno may have added receipts for some entries
298 Continue,
299
300 // Occasional result. The new seqno was at the end of the sequence (or
301 // an attempt at retrieving a trailing supporting signature), but we
302 // still have receiptless entries, so attempt to fetch the next
303 FetchNext,
304 };
305
306 std::optional<ccf::SeqNo> populate_receipts(ccf::SeqNo new_seqno)
307 {
308 HISTORICAL_LOG(
309 "Looking at {}, and populating receipts from it", new_seqno);
310 auto new_details = get_store_details(new_seqno);
311 if (new_details->store != nullptr)
312 {
313 if (new_details->is_signature)
314 {
315 HISTORICAL_LOG("{} is a signature", new_seqno);
316 // Iterate through earlier indices. If this signature covers them
317 // then create a receipt for them
318 const auto sig = get_signature(new_details->store);
319 ccf::MerkleTreeHistory tree(get_tree(new_details->store).value());
320
321 for (auto seqno : requested_seqnos)
322 {
323 if (seqno >= new_seqno)
324 {
325 break;
326 }
327
328 if (tree.in_range(seqno))
329 {
330 auto details = get_store_details(seqno);
331 if (details != nullptr && details->store != nullptr)
332 {
333 auto proof = tree.get_proof(seqno);
334 details->transaction_id = {sig->view, seqno};
335 details->receipt = std::make_shared<TxReceiptImpl>(
336 sig->sig,
337 proof.get_root(),
338 proof.get_path(),
339 sig->node,
340 sig->cert,
341 details->entry_digest,
342 details->get_commit_evidence(),
343 details->claims_digest);
344 HISTORICAL_LOG(
345 "Assigned a sig for {} after given signature at {}",
346 seqno,
347 new_seqno);
348 }
349 }
350 }
351 }
352 else
353 {
354 HISTORICAL_LOG("{} is not a signature", new_seqno);
355 const auto sig_it = supporting_signatures.find(new_seqno);
356 if (sig_it != supporting_signatures.end())
357 {
358 // This was a search for a supporting signature, but this entry is
359 // _not_ a signature - fetch the next
360 // NB: We skip any entries we already have here. It is possible we
361 // are fetching 10, previously had entries at 13, 14, 15, and the
362 // signature for all of these is at 20. The supporting signature
363 // for 10 tries 11, then 12. Next, it should try 16, not 13.
364 auto next_seqno = new_seqno + 1;
365 while (requested_seqnos.contains(next_seqno))
366 {
367 ++next_seqno;
368 }
369 HISTORICAL_LOG(
370 "{} was a supporting signature attempt, fetch next {}",
371 new_seqno,
372 next_seqno);
373 return {next_seqno};
374 }
375 else if (new_details->receipt == nullptr)
376 {
377 HISTORICAL_LOG(
378 "{} also has no receipt - looking for later signature",
379 new_seqno);
380 // Iterate through later indices, see if there's a signature that
381 // covers this one
382 const auto& untrusted_digest = new_details->entry_digest;
383 bool sig_seen = false;
384 std::optional<ccf::SeqNo> end_of_matching_range = std::nullopt;
385 for (const auto& [first_seqno, additional] :
386 requested_seqnos.get_ranges())
387 {
388 if (first_seqno + additional < new_seqno)
389 {
390 HISTORICAL_LOG(
391 "Ignoring range starting at {} - too early", first_seqno);
392 continue;
393 }
394
395 if (!end_of_matching_range.has_value())
396 {
397 end_of_matching_range = first_seqno + additional;
398 }
399
400 for (auto seqno = first_seqno;
401 seqno <= first_seqno + additional;
402 ++seqno)
403 {
404 if (seqno <= new_seqno)
405 {
406 HISTORICAL_LOG("Ignoring {} - too early", seqno);
407 continue;
408 }
409
410 auto details = get_store_details(seqno);
411 if (details != nullptr)
412 {
413 if (details->store != nullptr && details->is_signature)
414 {
415 const auto sig = get_signature(details->store);
416 ccf::MerkleTreeHistory tree(
417 get_tree(details->store).value());
418 if (tree.in_range(new_seqno))
419 {
420 auto proof = tree.get_proof(new_seqno);
421 new_details->transaction_id = {sig->view, new_seqno};
422 new_details->receipt = std::make_shared<TxReceiptImpl>(
423 sig->sig,
424 proof.get_root(),
425 proof.get_path(),
426 sig->node,
427 sig->cert,
428 new_details->entry_digest,
429 details->get_commit_evidence(),
430 new_details->claims_digest);
431 return std::nullopt;
432 }
433
434 // Break here - if this signature doesn't cover us, no
435 // later one can
436 sig_seen = true;
437 HISTORICAL_LOG(
438 "Found a sig for {} at {}", new_seqno, seqno);
439 break;
440 }
441 }
442 }
443
444 if (sig_seen)
445 {
446 break;
447 }
448 }
449
450 if (!sig_seen)
451 {
452 auto sig_it = supporting_signatures.lower_bound(new_seqno);
453 if (sig_it != supporting_signatures.end())
454 {
455 const auto& [sig_seqno, details] = *sig_it;
456 HISTORICAL_LOG(
457 "Considering a supporting signature for {} at {}",
458 new_seqno,
459 sig_seqno);
460 if (details->store != nullptr && details->is_signature)
461 {
462 const auto sig = get_signature(details->store);
463 ccf::MerkleTreeHistory tree(
464 get_tree(details->store).value());
465 if (tree.in_range(new_seqno))
466 {
467 auto proof = tree.get_proof(new_seqno);
468 new_details->transaction_id = {sig->view, new_seqno};
469 new_details->receipt = std::make_shared<TxReceiptImpl>(
470 sig->sig,
471 proof.get_root(),
472 proof.get_path(),
473 sig->node,
474 sig->cert,
475 new_details->entry_digest,
476 details->get_commit_evidence(),
477 new_details->claims_digest);
478 }
479 }
480 }
481 }
482
483 // If still have no receipt, after considering every larger value
484 // we have, and the best-guess at a supporting signature, then we
485 // may need to fetch another supporting signature. Request the
486 // first entry after the range
487 if (
488 new_details->receipt == nullptr &&
489 end_of_matching_range.has_value())
490 {
491 HISTORICAL_LOG(
492 "Still nothing, better fetch {}",
493 end_of_matching_range.value() + 1);
494 return {end_of_matching_range.value() + 1};
495 }
496 }
497 }
498 }
499
500 return std::nullopt;
501 }

514 void fetch_entry_at(ccf::SeqNo seqno)
515 {

518
519 void fetch_entries_range(ccf::SeqNo from, ccf::SeqNo to)
520 {
521 std::optional<ccf::SeqNo> unfetched_from = std::nullopt;
522 std::optional<ccf::SeqNo> unfetched_to = std::nullopt;
523
524 for (auto seqno = from; seqno <= to; ++seqno)
525 {
526 const auto ib = pending_fetches.insert(seqno);
527 if (ib.second)
528 {
529 if (!unfetched_from.has_value())
530 {
531 unfetched_from = seqno;
532 }
533 unfetched_to = seqno;
534 }
535 }
536
537 if (unfetched_from.has_value())
538 {
539 // Newly requested seqnos
540 RINGBUFFER_WRITE_MESSAGE(
541 consensus::ledger_get_range,
542 to_host,
543 static_cast<consensus::Index>(unfetched_from.value()),
544 static_cast<consensus::Index>(unfetched_to.value()),
545 consensus::LedgerRequestPurpose::HistoricalQuery);
546 }

549 std::unique_ptr<LedgerSecretRecoveryInfo> fetch_supporting_secret_if_needed(
550 ccf::SeqNo seqno)
551 {
552 auto [earliest_ledger_secret_seqno, earliest_ledger_secret] =
553 get_earliest_known_ledger_secret();
554 if (seqno < earliest_ledger_secret_seqno)
555 {
556 // Still need more secrets, fetch the next
557 auto previous_secret_stored_version =
558 earliest_ledger_secret->previous_secret_stored_version;
559 if (!previous_secret_stored_version.has_value())
560 {
561 throw std::logic_error(fmt::format(
562 "Earliest known ledger secret at {} has no earlier secret stored "
563 "version ({})",
564 earliest_ledger_secret_seqno,
565 seqno));
566 }
567
568 const auto seqno_to_fetch = previous_secret_stored_version.value();
569 LOG_TRACE_FMT(
570 "Requesting historical entry at {} but first known ledger "
571 "secret is applicable from {} - requesting older secret now",
572 seqno,
573 earliest_ledger_secret_seqno);
574
575 fetch_entry_at(seqno_to_fetch);
576 return std::make_unique<LedgerSecretRecoveryInfo>(
577 seqno_to_fetch, earliest_ledger_secret);
578 }
579
580 return nullptr;
581 }

583 void process_deserialised_store(
584 const kv::StorePtr& store,
585 const crypto::Sha256Hash& entry_digest,
586 ccf::SeqNo seqno,
587 bool is_signature,
588 ccf::ClaimsDigest&& claims_digest,
589 bool has_commit_evidence)
590 {
591 auto request_it = requests.begin();
592 while (request_it != requests.end())
593 {
594 auto& [handle, request] = *request_it;
595
596 // If this request was still waiting for a ledger secret, and this is
597 // that secret
598 if (
599 request.ledger_secret_recovery_info != nullptr &&
600 request.ledger_secret_recovery_info->target_seqno == seqno)
601 {
602 // Handle it, hopefully extending earliest_known_ledger_secret to
603 // cover earlier entries
604 const auto valid_secret = handle_encrypted_past_ledger_secret(
605 store, std::move(request.ledger_secret_recovery_info));
606 if (!valid_secret)
607 {
608 // Invalid! Erase this request: host gave us junk, need to start
609 // over
610 request_it = requests.erase(request_it);
611 continue;
612 }
613
614 auto new_secret_fetch =
615 fetch_supporting_secret_if_needed(request.first_requested_seqno);
616 if (new_secret_fetch != nullptr)
617 {
618 request.ledger_secret_recovery_info = std::move(new_secret_fetch);
619 }
620 else
621 {
622 // Newly have all required secrets - begin fetching the actual
623 // entries
624 for (const auto& [first_requested_seqno, num_following] :
625 request.requested_seqnos.get_ranges())
626 {
627 fetch_entries_range(
628 first_requested_seqno, first_requested_seqno + num_following);
629 }
630 }
631
632 // In either case, done with this request, try the next
633 ++request_it;
634 continue;
635 }
636
637 auto details = request.get_store_details(seqno);
638 if (
639 details != nullptr &&
640 details->current_stage == RequestStage::Fetching)
641 {
642 // Deserialisation includes a GCM integrity check, so all entries have
643 // been verified by the time we get here.
644 details->current_stage = RequestStage::Trusted;
645 details->has_commit_evidence = has_commit_evidence;
646
647 details->entry_digest = entry_digest;
648 if (!claims_digest.empty())
649 details->claims_digest = std::move(claims_digest);
650
651 CCF_ASSERT_FMT(
652 details->store == nullptr,
653 "Request {} already has store for seqno {}",
654 handle,
655 seqno);
656 details->store = store;
657
658 details->is_signature = is_signature;
659 if (is_signature)
660 {
661 // Construct a signature receipt.
662 // We do this whether it was requested or not, because we have all
663 // the state to do so already, and it's simpler than constructing
664 // the receipt _later_ for an already-fetched signature transaction.
665 const auto sig = get_signature(details->store);
666 assert(sig.has_value());
667 details->transaction_id = {sig->view, sig->seqno};
668 details->receipt = std::make_shared<TxReceiptImpl>(
669 sig->sig, sig->root.h, nullptr, sig->node, sig->cert);
670 }
671
672 if (request.include_receipts)
673 {
674 const auto next_seqno = request.populate_receipts(seqno);
675 if (next_seqno.has_value())
676 {
677 request.supporting_signatures.erase(seqno);
678 fetch_entry_at(*next_seqno);
679 request.supporting_signatures[*next_seqno] =
680 std::make_shared<StoreDetails>();
681 }
682 ++request_it;
683 }
684 }
685 else
686 {
687 ++request_it;
688 }
689 }
690 }

692 bool handle_encrypted_past_ledger_secret(
693 const kv::StorePtr& store,
694 std::unique_ptr<LedgerSecretRecoveryInfo> ledger_secret_recovery_info)
695 {
696 // Read encrypted secrets from store
697 auto tx = store->create_read_only_tx();
698 auto encrypted_past_ledger_secret =
699 tx.ro<ccf::EncryptedLedgerSecretsInfo>(
700 ccf::Tables::ENCRYPTED_PAST_LEDGER_SECRET);
701 if (!encrypted_past_ledger_secret)
702 {
703 return false;
704 }
705
706 // Construct description and decrypted secret
707 auto previous_ledger_secret =
708 encrypted_past_ledger_secret->get()->previous_ledger_secret;
709
710 auto recovered_ledger_secret = std::make_shared<LedgerSecret>(
711 ccf::decrypt_previous_ledger_secret_raw(
712 ledger_secret_recovery_info->last_ledger_secret,
713 std::move(previous_ledger_secret->encrypted_data)),
714 previous_ledger_secret->previous_secret_stored_version);
715
716 // Add recovered secret to historical secrets
717 historical_ledger_secrets->set_secret(
718 previous_ledger_secret->version, std::move(recovered_ledger_secret));
719
720 return true;
721 }

738 std::vector<StatePtr> get_states_internal(
739 const CompoundHandle& handle,
740 const SeqNoCollection& seqnos,
741 ExpiryDuration seconds_until_expiry,
742 bool include_receipts)
743 {
744 if (seqnos.empty())
745 {
746 throw std::logic_error(
747 "Invalid range for historical query: Cannot request empty range");
748 }
749
750 std::lock_guard<ccf::Mutex> guard(requests_lock);
751
752 const auto ms_until_expiry =
753 std::chrono::duration_cast<std::chrono::milliseconds>(
754 seconds_until_expiry);
755
756 auto it = requests.find(handle);
757 if (it == requests.end())
758 {
759 // This is a new handle - insert a newly created Request for it
760 it = requests.emplace_hint(it, handle, Request());
761 HISTORICAL_LOG("First time I've seen handle {}", handle);
762 }
763
764 Request& request = it->second;
765
766 // Update this Request to represent the currently requested ranges,
767 // returning any newly requested indices
768 auto new_seqnos = request.adjust_ranges(seqnos, include_receipts);
769
770 // If the earliest target entry cannot be deserialised with the earliest
771 // known ledger secret, record the target seqno and begin fetching the
772 // previous historical ledger secret.
773 auto secret_fetch =
774 fetch_supporting_secret_if_needed(request.first_requested_seqno);
775 if (secret_fetch != nullptr)
776 {
777 if (
778 request.ledger_secret_recovery_info == nullptr ||
779 request.ledger_secret_recovery_info->target_seqno !=
780 secret_fetch->target_seqno)
781 {
782 request.ledger_secret_recovery_info = std::move(secret_fetch);
783 }
784 }
785 else
786 {
787 // If we have sufficiently early secrets, begin fetching any newly
788 // requested entries. If we don't fall into this branch, they'll only
789 // begin to be fetched once the secret arrives.
790 for (const auto& [start_seqno, additional] : new_seqnos.get_ranges())
791 {
792 fetch_entries_range(start_seqno, start_seqno + additional);
793 }
794 }
795
796 // Reset the expiry timer as this has just been requested
797 request.time_to_expiry = ms_until_expiry;
798
799 std::vector<StatePtr> trusted_states;
800
801 for (auto seqno : seqnos)
802 {
803 auto target_details = request.get_store_details(seqno);
804 if (target_details == nullptr)
805 {
806 throw std::logic_error("Request isn't tracking state for seqno");
807 }
808
809 if (
810 target_details->current_stage == RequestStage::Trusted &&
811 (!request.include_receipts || target_details->receipt != nullptr))
812 {
813 // Have this store, associated txid and receipt and trust it - add
814 // it to return list
815 StatePtr state = std::make_shared<State>(
816 target_details->store,
817 target_details->receipt,
818 target_details->transaction_id);
819 trusted_states.push_back(state);
820 }
821 else
822 {
823 // Still fetching this store or don't trust it yet, so range is
824 // incomplete - return empty vector
825 return {};
826 }
827 }
828
829 return trusted_states;
830 }

1007 bool handle_ledger_entry(ccf::SeqNo seqno, const uint8_t* data, size_t size)
1008 {
1009 std::lock_guard<ccf::Mutex> guard(requests_lock);
1010 const auto it = pending_fetches.find(seqno);
1011 if (it == pending_fetches.end())
1012 {
1013 // Unexpected entry - ignore it?
1014 return false;
1015 }
1016
1017 pending_fetches.erase(it);
1018
1019 kv::ApplyResult deserialise_result;
1020 ccf::ClaimsDigest claims_digest;
1021 bool has_commit_evidence;
1022 auto store = deserialise_ledger_entry(
1023 seqno,
1024 data,
1025 size,
1026 deserialise_result,
1027 claims_digest,
1028 has_commit_evidence);
1029
1030 if (deserialise_result == kv::ApplyResult::FAIL)
1031 {
1032 return false;
1033 }
1034
1035 {
1036 // Confirm this entry is from a precursor of the current state, and not
1037 // a fork
1038 const auto tx_id = store->current_txid();
1039 if (tx_id.version != seqno)
1040 {
1041 LOG_FAIL_FMT(
1042 "Corrupt ledger entry received - claims to be {} but is actually "
1043 "{}.{}",
1044 seqno,
1045 tx_id.term,
1046 tx_id.version);
1047 return false;
1048 }
1049
1050 auto consensus = source_store.get_consensus();
1051 if (consensus == nullptr)
1052 {
1053 LOG_FAIL_FMT("No consensus on source store");
1054 return false;
1055 }
1056
1057 const auto actual_view = consensus->get_view(seqno);
1058 if (actual_view != tx_id.term)
1059 {
1060 LOG_FAIL_FMT(
1061 "Ledger entry comes from fork - contains {}.{} but this service "
1062 "expected {}.{}",
1063 tx_id.term,
1064 tx_id.version,
1065 actual_view,
1066 seqno);
1067 return false;
1068 }
1069 }
1070
1071 const auto is_signature =
1072 deserialise_result == kv::ApplyResult::PASS_SIGNATURE;
1073
1074 HISTORICAL_LOG(
1075 "Processing historical store at {} ({})",
1076 seqno,
1077 (size_t)deserialise_result);
1078 const auto entry_digest = crypto::Sha256Hash({data, size});
1079 process_deserialised_store(
1080 store,
1081 entry_digest,
1082 seqno,
1083 is_signature,
1084 std::move(claims_digest),
1085 has_commit_evidence);
1086
1087 return true;
1088 }

1097 bool handle_ledger_entries(
1098 ccf::SeqNo from_seqno,
1099 ccf::SeqNo to_seqno,
1100 const uint8_t* data,
1101 size_t size)
1102 {
1103 auto seqno = from_seqno;
1104 bool all_accepted = true;
1105 while (size > 0)
1106 {
1107 const auto header =
1108 serialized::peek<kv::SerialisedEntryHeader>(data, size);
1109 const auto whole_size = header.size + kv::serialised_entry_header_size;
1110 all_accepted &= handle_ledger_entry(seqno, data, whole_size);
1111 data += whole_size;
1112 size -= whole_size;
1113 ++seqno;
1114 }
1115
1116 CCF_ASSERT_FMT(
1117 seqno == to_seqno + 1,
1118 "Ledger entry range doesn't contain claimed entries");

1147 kv::StorePtr deserialise_ledger_entry(
1148 ccf::SeqNo seqno,
1149 const uint8_t* data,
1150 size_t size,
1151 kv::ApplyResult& result,
1152 ccf::ClaimsDigest& claims_digest,
1153 bool& has_commit_evidence)
1154 {
1155 // Create a new store and try to deserialise this entry into it
1156 kv::StorePtr store = std::make_shared<kv::Store>(
1157 false /* Do not start from very first seqno */,
1158 true /* Make use of historical secrets */);
1159
1160 // If this is older than the node's currently known ledger secrets, use
1161 // the historical encryptor (which should have older secrets)
1162 if (seqno < source_ledger_secrets->get_first().first)
1163 {
1164 store->set_encryptor(historical_encryptor);
1165 }
1166 else
1167 {
1168 store->set_encryptor(source_store.get_encryptor());
1169 }
1170
1171 try
1172 {
1173 // Encrypted ledger secrets are deserialised in public-only mode. Their
1174 // Merkle tree integrity is not verified: even if the recovered ledger
1175 // secret was bogus, the deserialisation of subsequent ledger entries
1176 // would fail.
1177 bool public_only = false;
1178 for (const auto& [_, request] : requests)
1179 {
1180 if (
1181 request.ledger_secret_recovery_info != nullptr &&
1182 request.ledger_secret_recovery_info->target_seqno == seqno)
1183 {
1184 public_only = true;
1185 break;
1186 }
1187 }
1188
1189 auto exec = store->deserialize(
1190 {data, data + size}, ConsensusType::CFT, public_only);
1191 if (exec == nullptr)
1192 {
1193 result = kv::ApplyResult::FAIL;
1194 return nullptr;
1195 }
1196
1197 result = exec->apply();
1198 claims_digest = std::move(exec->consume_claims_digest());
1199
1200 auto commit_evidence_digest =
1201 std::move(exec->consume_commit_evidence_digest());
1202 has_commit_evidence = commit_evidence_digest.has_value();
1203 }
1204 catch (const std::exception& e)
1205 {
1206 LOG_FAIL_FMT(
1207 "Exception while attempting to deserialise entry {}: {}",
1208 seqno,
1209 e.what());
1210 result = kv::ApplyResult::FAIL;
1211 }
1212
1213 return store;
1214 }

1216 void tick(const std::chrono::milliseconds& elapsed_ms)
1217 {
1218 std::lock_guard<ccf::Mutex> guard(requests_lock);
1219 auto it = requests.begin();
1220 while (it != requests.end())
1221 {
1222 auto& request = it->second;
1223 if (elapsed_ms >= request.time_to_expiry)
1224 {
1225 it = requests.erase(it);
1226 }
1227 else
1228 {
1229 request.time_to_expiry -= elapsed_ms;
1230 ++it;
1231 }
1232 }
