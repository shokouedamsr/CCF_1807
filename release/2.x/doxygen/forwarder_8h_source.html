53 bool forward_command(
54 std::shared_ptr<ccf::RpcContextImpl> rpc_ctx,
55 const NodeId& to,
56 const std::vector<uint8_t>& caller_cert) override
57 {
58 IsCallerCertForwarded include_caller = false;
59 const auto method = rpc_ctx->get_method();
60 const auto& raw_request = rpc_ctx->get_serialised_request();
61 size_t size = sizeof(rpc_ctx->get_session_context()->client_session_id) +
62 sizeof(IsCallerCertForwarded) + raw_request.size();
63 if (!caller_cert.empty())
64 {
65 size += sizeof(size_t) + caller_cert.size();
66 include_caller = true;
67 }
68
69 std::vector<uint8_t> plain(size);
70 auto data_ = plain.data();
71 auto size_ = plain.size();
72 serialized::write(
73 data_, size_, rpc_ctx->get_session_context()->client_session_id);
74 serialized::write(data_, size_, include_caller);
75 if (include_caller)
76 {
77 serialized::write(data_, size_, caller_cert.size());
78 serialized::write(data_, size_, caller_cert.data(), caller_cert.size());
79 }
80 serialized::write(data_, size_, raw_request.data(), raw_request.size());
81
82 ForwardedHeader msg = {
83 ForwardedMsg::forwarded_cmd, rpc_ctx->frame_format()};
84
85 return n2n_channels->send_encrypted(
86 to, NodeMsgType::forwarded_msg, plain, msg);
87 }

89 std::shared_ptr<http::HttpRpcContext> recv_forwarded_command(
90 const NodeId& from, const uint8_t* data, size_t size)
91 {
92 std::pair<ForwardedHeader, std::vector<uint8_t>> r;
93 try
94 {
95 LOG_TRACE_FMT("Receiving forwarded command of {} bytes", size);
96 LOG_TRACE_FMT(" => {:02x}", fmt::join(data, data + size, ""));
97
98 r = n2n_channels->template recv_encrypted<ForwardedHeader>(
99 from, data, size);
100 }
101 catch (const std::logic_error& err)
102 {
103 LOG_FAIL_FMT("Invalid forwarded command");
104 LOG_DEBUG_FMT("Invalid forwarded command: {}", err.what());
105 return nullptr;
106 }
107
108 std::vector<uint8_t> caller_cert;
109 const auto& plain_ = r.second;
110 auto data_ = plain_.data();
111 auto size_ = plain_.size();
112 auto client_session_id = serialized::read<size_t>(data_, size_);
113 auto includes_caller =
114 serialized::read<IsCallerCertForwarded>(data_, size_);
115 if (includes_caller)
116 {
117 auto caller_size = serialized::read<size_t>(data_, size_);
118 caller_cert = serialized::read(data_, size_, caller_size);
119 }
120 std::vector<uint8_t> raw_request = serialized::read(data_, size_, size_);
121
122 auto session =
123 std::make_shared<ccf::SessionContext>(client_session_id, caller_cert);
124 session->is_forwarded = true;
125
126 try
127 {
128 return ccf::make_fwd_rpc_context(
129 session, raw_request, r.first.frame_format);
130 }
131 catch (const std::exception& err)
132 {
133 LOG_FAIL_FMT("Invalid forwarded request");
134 LOG_DEBUG_FMT("Invalid forwarded request: {}", err.what());
135 return nullptr;
136 }
137 }

140 size_t client_session_id,

143 {

146 auto size_ = plain.size();

187 void recv_message(const ccf::NodeId& from, const uint8_t* data, size_t size)
188 {
189 try
190 {
191 auto forwarded_msg = serialized::peek<ForwardedMsg>(data, size);
192 LOG_TRACE_FMT(
193 "recv_message({}, {} bytes) (type={})",
194 from,
195 size,
196 (size_t)forwarded_msg);
197
198 switch (forwarded_msg)
199 {
200 case ForwardedMsg::forwarded_cmd:
201 {
202 std::shared_ptr<ccf::RPCMap> rpc_map_shared = rpc_map.lock();
203 if (rpc_map_shared)
204 {
205 auto ctx = recv_forwarded_command(from, data, size);
206 if (ctx == nullptr)
207 {
208 LOG_FAIL_FMT("Failed to receive forwarded command");
209 return;
210 }
211
212 const auto actor_opt = http::extract_actor(*ctx);
213 if (!actor_opt.has_value())
214 {
215 LOG_FAIL_FMT("Failed to extract actor from forwarded context.");
216 LOG_DEBUG_FMT(
217 "Failed to extract actor from forwarded context. Method is "
218 "'{}'",
219 ctx->get_method());
220 }
221
222 const auto& actor_s = actor_opt.value();
223 auto actor = rpc_map_shared->resolve(actor_s);
224 auto handler = rpc_map_shared->find(actor);
225 if (actor == ccf::ActorsType::unknown || !handler.has_value())
226 {
227 LOG_FAIL_FMT(
228 "Failed to process forwarded command: unknown actor");
229 LOG_DEBUG_FMT(
230 "Failed to process forwarded command: unknown actor {}",
231 actor_s);
232 return;
233 }
234
235 auto fwd_handler =
236 dynamic_cast<ForwardedRpcHandler*>(handler.value().get());
237 if (!fwd_handler)
238 {
239 LOG_FAIL_FMT(
240 "Failed to process forwarded command: handler is not a "
241 "ForwardedRpcHandler");
242 return;
243 }
244
245 // Ignore return value - false only means it is pending
246 send_forwarded_response(
247 ctx->get_session_context()->client_session_id,
248 from,
249 fwd_handler->process_forwarded(ctx));
250 LOG_DEBUG_FMT("Sending forwarded response to {}", from);
251 }
252 break;
253 }
254
255 case ForwardedMsg::forwarded_response:
256 {
257 auto rep = recv_forwarded_response(from, data, size);
258 if (!rep.has_value())
259 {
260 return;
261 }
262
263 LOG_DEBUG_FMT(
264 "Sending forwarded response to RPC endpoint {}", rep->first);
265
266 auto rpc_responder_shared = rpcresponder.lock();
267 if (
268 rpc_responder_shared &&
269 !rpc_responder_shared->reply_async(
270 rep->first, std::move(rep->second)))
271 {
272 return;
273 }
274
275 break;
276 }
277
278 default:
279 {
280 LOG_FAIL_FMT("Unknown frontend msg type: {}", forwarded_msg);
281 break;
282 }
283 }
284 }
285 catch (const std::exception& e)
286 {
287 LOG_FAIL_EXC(e.what());
288 return;
289 }
290 }
291 };
