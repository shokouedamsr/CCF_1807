106 template <typename Serializer, typename... Ts>
107 bool write_multiple(Message m, bool wait, Ts&&... ts)
108 {
109 auto sections = Serializer::serialize(std::forward<Ts>(ts)...);
110
111 // Fold section->sizes over the + operator, with initial value 0
112 size_t total_size = std::apply(
113 [](const auto&... section) { return (section->size() + ... + 0); },
114 sections);
115
116 const auto initial_marker = prepare(m, total_size, wait);
117
118 if (!initial_marker.has_value())
119 return false;
120
121 auto next = initial_marker;
122 serializer::details::tuple_for_each(sections, [&](const auto& s) {
123 next = write_bytes(next, s->data(), s->size());
124 });
125
126 finish(initial_marker);
127
128 return next.has_value();
129 }
