83 msg->data.self = this->shared_from_this();

90 void recv_(const uint8_t* data, size_t size)
91 {
92 recv_buffered(data, size);
93
94 if (get_status() == Status::ready)
95 {
96 // Send HTTP response describing soft session limit
97 auto http_response = http::Response(HTTP_STATUS_SERVICE_UNAVAILABLE);
98
99 http_response.set_header(
100 http::headers::CONTENT_TYPE, http::headervalues::contenttype::JSON);
101
102 nlohmann::json body = ccf::ODataErrorResponse{ccf::ODataError{
103 ccf::errors::SessionCapExhausted,
104 "Service is currently busy and unable to serve new connections"}};
105 const auto s = body.dump();
106 http_response.set_body((const uint8_t*)s.data(), s.size());
107
108 send(http_response.build_response());
109
110 // Close connection
111 close();
112 }
113 }

121 tls::ConnID get_next_client_id()
122 {
123 auto id = next_client_session_id--;
124 const auto initial = id;
125
126 if (next_client_session_id > 0)
127 next_client_session_id = -1;
128
129 while (sessions.find(id) != sessions.end())
130 {
131 id--;
132
133 if (id > 0)
134 id = -1;
135
136 if (id == initial)
137 {
138 throw std::runtime_error(
139 "Exhausted all IDs for enclave client sessions");
140 }
141 }
142
143 return id;
144 }

146 ListenInterface& get_interface_from_session_id(tls::ConnID id)
147 {
148 // Lock must be first acquired and held while accessing returned interface
149 auto search = sessions.find(id);
150 if (search != sessions.end())
151 {
152 auto it = listening_interfaces.find(search->second.first);
153 if (it != listening_interfaces.end())
154 {
155 return it->second;
156 }
157 }
158
159 throw std::logic_error(
160 fmt::format("No RPC interface for session ID {}", id));
161 }

191 void update_listening_interface_options(
192 const ccf::NodeInfoNetwork& node_info)
193 {
194 std::lock_guard<ccf::Mutex> guard(lock);
195
196 for (const auto& [name, interface] : node_info.rpc_interfaces)
197 {
198 auto& li = listening_interfaces[name];
199
200 li.max_open_sessions_soft = interface.max_open_sessions_soft.value_or(
201 max_open_sessions_soft_default);
202
203 li.max_open_sessions_hard = interface.max_open_sessions_hard.value_or(
204 max_open_sessions_hard_default);
205
206 li.endorsement = interface.endorsement.value_or(endorsement_default);
207
208 li.http_configuration =
209 interface.http_configuration.value_or(http::ParserConfiguration{});
210
211 LOG_INFO_FMT(
212 "Setting max open sessions on interface \"{}\" ({}) to [{}, "
213 "{}] and endorsement authority to {}",
214 name,
215 interface.bind_address,
216 li.max_open_sessions_soft,
217 li.max_open_sessions_hard,
218 li.endorsement.authority);
219 }

253 void set_cert(
254 ccf::Authority authority, const crypto::Pem& cert_, const crypto::Pem& pk)
255 {
256 // Caller authentication is done by each frontend by looking up
257 // the caller's certificate in the relevant store table. The caller
258 // certificate does not have to be signed by a known CA (nullptr) and
259 // verification is not required here.
260 auto cert = std::make_shared<tls::Cert>(
261 nullptr, cert_, pk, std::nullopt, /*auth_required ==*/false);
262
263 std::lock_guard<ccf::Mutex> guard(lock);
264
265 for (auto& [listen_interface_id, interface] : listening_interfaces)
266 {
267 if (interface.endorsement.authority == authority)
268 {
269 certs.insert_or_assign(listen_interface_id, cert);
270 }
271 }

274 void accept(tls::ConnID id, const ListenInterfaceID& listen_interface_id)
275 {
276 std::lock_guard<ccf::Mutex> guard(lock);
277
278 if (sessions.find(id) != sessions.end())
279 {
280 throw std::logic_error(
281 fmt::format("Duplicate conn ID received inside enclave: {}", id));
282 }
283
284 auto it = listening_interfaces.find(listen_interface_id);
285 if (it == listening_interfaces.end())
286 {
287 throw std::logic_error(fmt::format(
288 "Can't accept new RPC session {} - comes from unknown listening "
289 "interface {}",
290 id,
291 listen_interface_id));
292 }
293
294 auto& per_listen_interface = it->second;
295
296 if (certs.find(listen_interface_id) == certs.end())
297 {
298 LOG_DEBUG_FMT(
299 "Refusing TLS session {} inside the enclave - interface {} "
300 "has no TLS certificate yet",
301 id,
302 listen_interface_id);
303
304 RINGBUFFER_WRITE_MESSAGE(
305 tls::tls_stop, to_host, id, std::string("Session refused"));
306 }
307 else if (
308 per_listen_interface.open_sessions >=
309 per_listen_interface.max_open_sessions_hard)
310 {
311 LOG_INFO_FMT(
312 "Refusing TLS session {} inside the enclave - already have {} "
313 "sessions from interface {} and limit is {}",
314 id,
315 per_listen_interface.open_sessions,
316 listen_interface_id,
317 per_listen_interface.max_open_sessions_hard);
318
319 RINGBUFFER_WRITE_MESSAGE(
320 tls::tls_stop, to_host, id, std::string("Session refused"));
321 }
322 else if (
323 per_listen_interface.open_sessions >=
324 per_listen_interface.max_open_sessions_soft)
325 {
326 LOG_INFO_FMT(
327 "Soft refusing session {} (returning 503) inside the enclave - "
328 "already have {} sessions from interface {} and limit is {}",
329 id,
330 per_listen_interface.open_sessions,
331 listen_interface_id,
332 per_listen_interface.max_open_sessions_soft);
333
334 auto ctx = std::make_unique<tls::Server>(certs[listen_interface_id]);
335 auto capped_session = std::make_shared<NoMoreSessionsEndpointImpl>(
336 id, writer_factory, std::move(ctx));
337 sessions.insert(std::make_pair(
338 id, std::make_pair(listen_interface_id, std::move(capped_session))));
339 per_listen_interface.open_sessions++;
340 per_listen_interface.peak_sessions = std::max(
341 per_listen_interface.peak_sessions,
342 per_listen_interface.open_sessions);
343 }
344 else
345 {
346 LOG_DEBUG_FMT(
347 "Accepting a session {} inside the enclave from interface \"{}\"",
348 id,
349 listen_interface_id);
350 auto ctx = std::make_unique<tls::Server>(certs[listen_interface_id]);
351
352 auto session = std::make_shared<ServerEndpointImpl>(
353 rpc_map,
354 id,
355 listen_interface_id,
356 writer_factory,
357 std::move(ctx),
358 per_listen_interface.http_configuration,
359 shared_from_this());
360 sessions.insert(std::make_pair(
361 id, std::make_pair(listen_interface_id, std::move(session))));
362 per_listen_interface.open_sessions++;
363 per_listen_interface.peak_sessions = std::max(
364 per_listen_interface.peak_sessions,
365 per_listen_interface.open_sessions);
366 }
367
368 sessions_peak = std::max(sessions_peak, sessions.size());
369 }

371 bool reply_async(tls::ConnID id, std::vector<uint8_t>&& data) override
372 {
373 std::lock_guard<ccf::Mutex> guard(lock);
374
375 auto search = sessions.find(id);
376 if (search == sessions.end())
377 {
378 LOG_DEBUG_FMT("Refusing to reply to unknown session {}", id);
379 return false;
380 }
381
382 LOG_DEBUG_FMT("Replying to session {}", id);
383
384 search->second.second->send(std::move(data));
385 return true;
386 }

388 void remove_session(tls::ConnID id)
389 {
390 std::lock_guard<ccf::Mutex> guard(lock);
391 LOG_DEBUG_FMT("Closing a session inside the enclave: {}", id);
392 const auto search = sessions.find(id);
393 if (search != sessions.end())
394 {
395 auto it = listening_interfaces.find(search->second.first);
396 if (it != listening_interfaces.end())
397 {
398 it->second.open_sessions--;
399 }
400 sessions.erase(search);
401 }

404 std::shared_ptr<ClientEndpoint> create_client(
405 std::shared_ptr<tls::Cert> cert)
406 {
407 std::lock_guard<ccf::Mutex> guard(lock);
408 auto ctx = std::make_unique<tls::Client>(cert);
409 auto id = get_next_client_id();
410
411 LOG_DEBUG_FMT("Creating a new client session inside the enclave: {}", id);
412
413 auto session = std::make_shared<ClientEndpointImpl>(
414 id, writer_factory, std::move(ctx));
415
416 // There are no limits on outbound client sessions (we do not check any
417 // session caps here). We expect this type of session to be rare and want
418 // it to succeed even when we are busy.
419 sessions.insert(std::make_pair(id, std::make_pair("", session)));
420
421 sessions_peak = std::max(sessions_peak, sessions.size());
422
423 return session;
424 }

426 void register_message_handlers(
427 messaging::Dispatcher<ringbuffer::Message>& disp)
428 {
429 DISPATCHER_SET_MESSAGE_HANDLER(
430 disp, tls::tls_start, [this](const uint8_t* data, size_t size) {
431 auto [new_tls_id, listen_interface_name] =
432 ringbuffer::read_message<tls::tls_start>(data, size);
433 accept(new_tls_id, listen_interface_name);
434 });
435
436 DISPATCHER_SET_MESSAGE_HANDLER(
437 disp, tls::tls_inbound, [this](const uint8_t* data, size_t size) {
438 auto [id, body] =
439 ringbuffer::read_message<tls::tls_inbound>(data, size);
440
441 auto search = sessions.find(id);
442 if (search == sessions.end())
443 {
444 LOG_DEBUG_FMT(
445 "Ignoring tls_inbound for unknown or refused session: {}", id);
446 return;
447 }
448
449 search->second.second->recv(body.data, body.size);
450 });
451
452 DISPATCHER_SET_MESSAGE_HANDLER(
453 disp, tls::tls_close, [this](const uint8_t* data, size_t size) {
454 auto [id] = ringbuffer::read_message<tls::tls_close>(data, size);
455 remove_session(id);
456 });
457 }
