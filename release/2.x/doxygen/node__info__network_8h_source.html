124 // The JSON representation of a NodeInfoNetwork is the union of a
125 // NodeInfoNetwork_v1 and a NodeInfoNetwork_v2. It contains the fields of
126 // both, so can be read as (or from!) either
127 inline void to_json(nlohmann::json& j, const NodeInfoNetwork& nin)
128 {
129 {
130 NodeInfoNetwork_v1 v1;
131 std::tie(v1.nodehost, v1.nodeport) =
132 split_net_address(nin.node_to_node_interface.bind_address);
133
134 if (nin.rpc_interfaces.size() > 0)
135 {
136 const auto& primary_interface = nin.rpc_interfaces.begin()->second;
137 std::tie(v1.rpchost, v1.rpcport) =
138 split_net_address(primary_interface.bind_address);
139 std::tie(v1.pubhost, v1.pubport) =
140 split_net_address(primary_interface.published_address);
141 }
142 to_json(j, v1);
143 }
144
145 to_json(j, (const NodeInfoNetwork_v2&)nin);
146 }
147
148 inline void from_json(const nlohmann::json& j, NodeInfoNetwork& nin)
149 {
150 try
151 {
152 NodeInfoNetwork_v2 v2;
153 from_json(j, v2);
154 nin = NodeInfoNetwork(v2);
155 }
156 catch (const JsonParseError& jpe)
157 {
158 NodeInfoNetwork_v1 v1;
159 from_json(j, v1);
160
161 nin.node_to_node_interface.bind_address =
162 make_net_address(v1.nodehost, v1.nodeport);
163
164 NodeInfoNetwork::NetInterface primary_interface;
165 primary_interface.bind_address = make_net_address(v1.rpchost, v1.rpcport);
166 primary_interface.published_address =
167 make_net_address(v1.pubhost, v1.pubport);
168
169 nin.rpc_interfaces.emplace(
170 PRIMARY_RPC_INTERFACE, std::move(primary_interface));
171 }
