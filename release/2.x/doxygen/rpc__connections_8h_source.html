117 bool listen(
118 tls::ConnID id,
119 std::string& host,
120 std::string& service,
121 const std::string& name)
122 {
123 if (id == 0)
124 {
125 id = get_next_id();
126 }
127
128 if (sockets.find(id) != sockets.end())
129 {
130 LOG_FAIL_FMT("Cannot listen on id {}: already in use", id);
131 return false;
132 }
133
134 TCP s;
135 s->set_behaviour(std::make_unique<RPCServerBehaviour>(*this, id));
136
137 if (!s->listen(host, service, name))
138 {
139 return false;
140 }
141
142 host = s->get_host();
143 service = s->get_service();
144
145 sockets.emplace(id, s);
146 return true;
147 }

149 bool connect(
150 int64_t id, const std::string& host, const std::string& service)
151 {
152 if (id == 0)
153 {
154 id = get_next_id();
155 }
156
157 if (sockets.find(id) != sockets.end())
158 {
159 LOG_FAIL_FMT("Cannot connect on id {}: already in use", id);
160 return false;
161 }
162
163 auto s = TCP(true, client_connection_timeout);
164 s->set_behaviour(std::make_unique<RPCClientBehaviour>(*this, id));
165
166 if (!s->connect(host, service))
167 {
168 return false;
169 }
170
171 sockets.emplace(id, s);
172 return true;
173 }

175 bool write(int64_t id, size_t len, const uint8_t* data)
176 {
177 auto s = sockets.find(id);
178
179 if (s == sockets.end())
180 {
181 LOG_FAIL_FMT(
182 "Received an outbound message for id {} which is not a known "
183 "connection. Ignoring message of {} bytes",
184 id,
185 len);
186 return false;
187 }
188
189 if (s->second.is_null())
190 return false;
191
192 return s->second->write(len, data);
193 }

195 bool stop(int64_t id)
196 {
197 // Invalidating the TCP socket will result in the handle being closed. No
198 // more messages will be read from or written to the TCP socket.
199 sockets[id] = nullptr;

224 int64_t connect_id = (int64_t)id;
225 LOG_DEBUG_FMT("rpc write from enclave {}: {}", connect_id, body.size);
226
227 write(connect_id, body.size, body.data);
228 });
229
230 DISPATCHER_SET_MESSAGE_HANDLER(
231 disp, tls::tls_connect, [this](const uint8_t* data, size_t size) {
232 auto [id, host, service] =
233 ringbuffer::read_message<tls::tls_connect>(data, size);
234
235 LOG_DEBUG_FMT("rpc connect request from enclave {}", id);
236
237 if (check_enclave_side_id(id))
238 {
239 connect(id, host, service);
240 }
241 else
242 {
243 LOG_FAIL_FMT(
244 "rpc session id is not in dedicated from-enclave range ({})", id);
245 }
246 });

248 DISPATCHER_SET_MESSAGE_HANDLER(

252 LOG_DEBUG_FMT("rpc stop from enclave {}, {}", id, msg);

265 private:
266 tls::ConnID get_next_id()
267 {
268 auto id = next_id++;
269 const auto initial = id;
270
271 if (next_id < 0)
272 next_id = 1;
273
274 while (sockets.find(id) != sockets.end())
275 {
276 id++;
277
278 if (id < 0)
279 id = 1;
280
281 if (id == initial)
282 {
283 throw std::runtime_error(
284 "Exhausted all IDs for host RPC connections");
285 }
286 }
287
288 return id;
289 }
290
291 bool check_enclave_side_id(tls::ConnID id)
292 {
293 return id < 0;
294 }
295
296 std::string get_interface_listen_name(tls::ConnID id)
297 {
298 const auto it = sockets.find(id);
299 if (it == sockets.end())
300 {
301 throw std::logic_error(fmt::format("No socket with id {}", id));
302 }
303
304 auto listen_name = it->second->get_listen_name();
305 if (!listen_name.has_value())
306 {
307 throw std::logic_error(
308 fmt::format("Interface {} has no listen name", id));
309 }
310
311 return listen_name.value();
312 }
