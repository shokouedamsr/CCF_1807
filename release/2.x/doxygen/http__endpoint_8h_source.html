28 protected:

34 tls::ConnID session_id,

37 const std::shared_ptr<ErrorReporter>& error_reporter = nullptr) :
38 TLSEndpoint(session_id, writer_factory, std::move(ctx)),

42
43 public:
44 static void recv_cb(std::unique_ptr<threading::Tmsg<SendRecvMsg>> msg)
45 {
46 reinterpret_cast<HTTPEndpoint*>(msg->data.self.get())
47 ->recv_(msg->data.data.data(), msg->data.data.size());
48 }
49
50 void recv(const uint8_t* data, size_t size) override
51 {
52 auto msg = std::make_unique<threading::Tmsg<SendRecvMsg>>(&recv_cb);
53 msg->data.self = this->shared_from_this();
54 msg->data.data.assign(data, data + size);
55
56 threading::ThreadMessaging::thread_messaging.add_task(
57 execution_thread, std::move(msg));
58 }
59
60 void recv_(const uint8_t* data_, size_t size_)
61 {
62 recv_buffered(data_, size_);
63
64 LOG_TRACE_FMT("recv called with {} bytes", size_);
65
66 constexpr auto read_block_size = 4096;
67 std::vector<uint8_t> buf(read_block_size);
68 auto data = buf.data();
69 auto n_read = read(data, buf.size(), false);
70
71 while (true)
72 {
73 if (n_read == 0)
74 {
75 return;
76 }
77
78 LOG_TRACE_FMT("Going to parse {} bytes", n_read);
79
80 try
81 {
82 p.execute(data, n_read);
83
84 // Used all provided bytes - check if more are available
85 n_read = read(buf.data(), buf.size(), false);
86 }
87 catch (RequestPayloadTooLarge& e)
88 {
89 if (error_reporter)
90 {
91 error_reporter->report_request_payload_too_large_error(session_id);
92 }
93
94 LOG_DEBUG_FMT("Request is too large: {}", e.what());
95
96 send_raw(http::error(ccf::ErrorDetails{
97 HTTP_STATUS_PAYLOAD_TOO_LARGE,
98 ccf::errors::RequestBodyTooLarge,
99 e.what()}));
100
101 close();
102 break;
103 }
104 catch (RequestHeaderTooLarge& e)
105 {
106 if (error_reporter)
107 {
108 error_reporter->report_request_header_too_large_error(session_id);
109 }
110
111 LOG_DEBUG_FMT("Request header is too large: {}", e.what());
112
113 send_raw(http::error(ccf::ErrorDetails{
114 HTTP_STATUS_REQUEST_HEADER_FIELDS_TOO_LARGE,
115 ccf::errors::RequestHeaderTooLarge,
116 e.what()}));
117
118 close();
119 break;
120 }
121 catch (const std::exception& e)
122 {
123 if (error_reporter)
124 {
125 error_reporter->report_parsing_error(session_id);
126 }
127 LOG_DEBUG_FMT("Error parsing HTTP request: {}", e.what());
128
129 auto response = http::Response(HTTP_STATUS_BAD_REQUEST);
130 response.set_header(
131 http::headers::CONTENT_TYPE, http::headervalues::contenttype::TEXT);
132 // NB: Avoid formatting input data a string, as it may contain null
133 // bytes. Instead insert it at the end of this message, verbatim
134 auto body_s = fmt::format(
135 "Unable to parse data as a HTTP request. Error message is: {}\n"
136 "Error occurred while parsing fragment:\n",
137 e.what());
138 std::vector<uint8_t> response_body(
139 std::begin(body_s), std::end(body_s));
140 response_body.insert(response_body.end(), data, data + n_read);
141 response.set_body(response_body.data(), response_body.size());
142 send_raw(response.build_response());
143
144 close();
145 break;
146 }
147 }
148 }

188 void handle_request(
189 llhttp_method verb,
190 const std::string_view& url,
191 http::HeaderMap&& headers,
192 std::vector<uint8_t>&& body) override
193 {
194 LOG_TRACE_FMT(
195 "Processing msg({}, {} [{} bytes])",
196 llhttp_method_name(verb),
197 url,
198 body.size());
199
200 try
201 {
202 if (session_ctx == nullptr)
203 {
204 session_ctx = std::make_shared<ccf::SessionContext>(
205 session_id, peer_cert(), interface_id);
206 }
207
208 std::shared_ptr<http::HttpRpcContext> rpc_ctx = nullptr;
209 try
210 {
211 rpc_ctx = std::make_shared<HttpRpcContext>(
212 session_ctx, verb, url, std::move(headers), std::move(body));
213 }
214 catch (std::exception& e)
215 {
216 send_raw(http::error(
217 HTTP_STATUS_INTERNAL_SERVER_ERROR,
218 ccf::errors::InternalError,
219 e.what()));
220 }
221
222 const auto actor_opt = http::extract_actor(*rpc_ctx);
223 if (!actor_opt.has_value())
224 {
225 rpc_ctx->set_error(
226 HTTP_STATUS_NOT_FOUND,
227 ccf::errors::ResourceNotFound,
228 fmt::format(
229 "Request path must contain '/[actor]/[method]'. Unable to parse "
230 "'{}'.",
231 rpc_ctx->get_method()));
232 send_raw(rpc_ctx->serialise_response());
233 return;
234 }
235
236 const auto& actor_s = actor_opt.value();
237 auto actor = rpc_map->resolve(actor_s);
238 auto search = rpc_map->find(actor);
239 if (actor == ccf::ActorsType::unknown || !search.has_value())
240 {
241 rpc_ctx->set_error(
242 HTTP_STATUS_NOT_FOUND,
243 ccf::errors::ResourceNotFound,
244 fmt::format("Unknown actor '{}'.", actor_s));
245 send_raw(rpc_ctx->serialise_response());
246 return;
247 }
248
249 auto response = search.value()->process(rpc_ctx);
250
251 if (!response.has_value())
252 {
253 // If the RPC is pending, hold the connection.
254 LOG_TRACE_FMT("Pending");
255 return;
256 }
257 else
258 {
259 send_buffered(response.value());
260 flush();
261 }
262 }
263 catch (const std::exception& e)
264 {
265 send_raw(http::error(
266 HTTP_STATUS_INTERNAL_SERVER_ERROR,
267 ccf::errors::InternalError,
268 fmt::format("Exception: {}", e.what())));
269
270 // On any exception, close the connection.
271 LOG_FAIL_FMT("Closing connection");
272 LOG_DEBUG_FMT("Closing connection due to exception: {}", e.what());
273 close();
274 throw;
275 }
276 }
