20 auto signature_size = signature.size();

23 OpenSSL::Unique_BIGNUM s;

28 // Ignore previous pointers, as they're now managed by ECDSA_SIG_set0
29 // https://www.openssl.org/docs/man1.1.1/man3/ECDSA_SIG_get0.html
30 (void)r.release();

35 auto der_sig_buf = der_sig.data();
36 OpenSSL::CHECK0(i2d_ECDSA_SIG(sig, &der_sig_buf));
