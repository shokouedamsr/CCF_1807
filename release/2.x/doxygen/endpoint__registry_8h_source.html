40 inline std::optional<PathTemplateSpec> parse_path_template(
41 const std::string& uri)
42 {
43 auto template_start = uri.find_first_of('{');
44 if (template_start == std::string::npos)
45 {
46 return std::nullopt;
47 }
48
49 PathTemplateSpec spec;
50
51 std::string regex_s = uri;
52 template_start = regex_s.find_first_of('{');
53 while (template_start != std::string::npos)
54 {
55 const auto template_end = regex_s.find_first_of('}', template_start);
56 if (template_end == std::string::npos)
57 {
58 throw std::logic_error(fmt::format(
59 "Invalid templated path - missing closing curly bracket: {}", uri));
60 }
61
62 spec.template_component_names.push_back(
63 regex_s.substr(template_start + 1, template_end - template_start - 1));
64 regex_s.replace(
65 template_start, template_end - template_start + 1, "([^/]+)");
66 template_start = regex_s.find_first_of('{', template_start + 1);
67 }
68
69 LOG_TRACE_FMT("Parsed a templated endpoint: {} became {}", uri, regex_s);
70 LOG_TRACE_FMT(
71 "Component names are: {}",
72 fmt::join(spec.template_component_names, ", "));
73 spec.template_regex = std::regex(regex_s);
74
75 return spec;
76 }

112 template <typename T>
113 bool get_path_param(
114 const ccf::PathParams& params,
115 const std::string& param_name,
116 T& value,
117 std::string& error)
118 {
119 const auto it = params.find(param_name);
120 if (it == params.end())
121 {
122 error = fmt::format("No parameter named '{}' in path", param_name);
123 return false;
124 }
125
126 const auto param_s = it->second;
127 const auto [p, ec] =
128 std::from_chars(param_s.data(), param_s.data() + param_s.size(), value);
129 if (ec != std::errc())
130 {
131 error = fmt::format(
132 "Unable to parse path parameter '{}' as a {}", param_s, param_name);
133 return false;
134 }
135
136 return true;
137 }
