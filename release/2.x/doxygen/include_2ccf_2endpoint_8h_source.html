235 template <typename In, typename Out>
236 Endpoint& set_auto_schema(std::optional<http_status> status = std::nullopt)
237 {
238 if constexpr (!std::is_same_v<In, void>)
239 {
240 params_schema = ds::json::build_schema<In>();
241
242 schema_builders.push_back(
243 [](nlohmann::json& document, const Endpoint& endpoint) {
244 const auto http_verb = endpoint.dispatch.verb.get_http_method();
245 if (!http_verb.has_value())
246 {
247 // Non-HTTP endpoints are not documented
248 return;
249 }
250
251 ds::openapi::add_request_body_schema<In>(
252 document, endpoint.full_uri_path, http_verb.value());
253 });
254 }
255 else
256 {
257 params_schema = nullptr;
258 }
259
260 if constexpr (!std::is_same_v<Out, void>)
261 {
262 success_status = status.value_or(HTTP_STATUS_OK);
263
264 result_schema = ds::json::build_schema<Out>();
265
266 schema_builders.push_back(
267 [](nlohmann::json& document, const Endpoint& endpoint) {
268 const auto http_verb = endpoint.dispatch.verb.get_http_method();
269 if (!http_verb.has_value())
270 {
271 return;
272 }
273
274 ds::openapi::add_response_schema<Out>(
275 document,
276 endpoint.full_uri_path,
277 http_verb.value(),
278 endpoint.success_status);
279 });
280 }
281 else
282 {
283 success_status = status.value_or(HTTP_STATUS_NO_CONTENT);
284 result_schema = nullptr;
285 }
286
287 return *this;
288 }

321 template <typename T>
322 Endpoint& add_query_parameter(
323 const std::string& param_name,
324 QueryParamPresence presence = RequiredParameter)
325 {
326 schema_builders.push_back(
327 [param_name,
328 presence](nlohmann::json& document, const Endpoint& endpoint) {
329 const auto http_verb = endpoint.dispatch.verb.get_http_method();
330 if (!http_verb.has_value())
331 {
332 // Non-HTTP endpoints are not documented
333 return;
334 }
335
336 const auto schema_name = ds::json::schema_name<T>();
337 const auto query_schema = ds::json::build_schema<T>();
338
339 auto parameter = nlohmann::json::object();
340 parameter["name"] = param_name;
341 parameter["in"] = "query";
342 parameter["required"] = presence == RequiredParameter;
343 parameter["schema"] = ds::openapi::add_schema_to_components(
344 document, schema_name, query_schema);
345 ds::openapi::add_request_parameter_schema(
346 document, endpoint.full_uri_path, http_verb.value(), parameter);
347 });
348
349 return *this;
350 }
